 11 iAccount = function (val) {
 12 	/**
 13 	 * set default
 14 	 * @type {null}
 15 	 */
 16 	this.val = null;
 17 	/**
 18 	 * @type {null}
 19 	 */
 20 	this.account = null;
 21 
 22 
 23 	//===================================
 24 	// param is undefined
 25 	if (typeof val === 'undefined') {
 26 		_error('iAccount: Account is undefined');
 27 
 28 		return;
 29 	} else {
 30 		/**
 31 		 * param is iAccount
 32  		 */
 33 		if (typeof val === 'object' && val instanceof iAccount) {
 34 
 35 			this.val = val.val;
 36 		} else {
 37 			/**
 38 			 * param is number
 39 			 */
 40 			if (typeof val === 'number') {
 41 
 42 				this.val = val.toString();
 43 			} else if (typeof val === 'string') {
 44 				var arr = val.split('-'); // для вида number-account
 45 				val = arr[0];
 46 
 47 				this.val = val.replace(/[^0-9]+/g, '') || null;
 48 
 49 				// TODO: remove. Used at XSLT JS
 50 				this.account = (arr[1] || '').replace(/[^0-9]+/g, '') || this.val;
 51 
 52 			}
 53 		}
 54 	}
 55 
 56     if (this.account === null) {
 57         this.account = this.val;
 58 	}
 59 
 60 	/**
 61 	 * check data
 62 	 */
 63 	if (this.validate()) {
 64 		return this;
 65 	}
 66 
 67 	/**
 68 	 * not parsed propertly
 69 	 */
 70 	_error('iAccount: Invalid account "' + val + '"');
 71 
 72 };

 74 iAccount.prototype = {
 75 
 76 	'val':	null,
 77 	countBeforeSplitter:	4,
 78 	minLength:	12,
 79 	splitter:	'<span>',
 80 
 81 	/**
 82 	 * add ' ' every [num] digits
 83 	 * @param str
 84 	 * @param splitter
 85 	 * @return {String}
 86 	 * @private
 87 	 */
 88 	_addSplitter: function (str, splitter) {
 89 
 90 		splitter = splitter || this.splitter;
 91 
 92 		/**
 93 		 * local copy
 94 		 * @type {*}
 95 		 */
 96 		var str = str;
 97 		var arr = [];
 98 
 99 		while (str.length) {
100 			if (str.length > this.countBeforeSplitter) {
101 				arr.push(str.substring(0, this.countBeforeSplitter));
102 				str = str.substring(this.countBeforeSplitter);
103 			} else {
104 				arr.push(str.substring(0, str.length));
105 				str = '';
106 			}
107 		}
108 
109 		/**
110 		 * splitter is tag?
111 		 */
112 		if ((/^<[^>]+>$/).test(splitter)) {
113 
114 			var openedTag = splitter;
115 			var closedTag = splitter.replace(/^<([^>]+)>$/, '</$1>');
116 
117 			return openedTag + arr.join(closedTag + openedTag) + closedTag;
118 		}
119 
120 		return arr.join(splitter);
121 	},
122 
123 	formatters:	{
124 
125 		/**
126 		 * For example, XXXX XXXX XXXX XXXX
127 		 * @param intStr
128 		 * @param splitter
129 		 * @return {String}
130 		 */
131 		full:	function(intStr, splitter) {
132 		
133 			return this._addSplitter(intStr, splitter);
134 		},
135 
136 		/**
137 		 * For example, XXXX **** **** XXXX
138 		 * @param intStr
139 		 * @param splitter
140 		 * @return {String}
141 		 */
142 		hidden:	function (intStr, splitter) {
143 		
144 			var hidden = '';
145 			var i = intStr.length - this.countBeforeSplitter * 2;
146 			while (i--) {
147 				hidden += '*';
148 			}
149 
150 			return this._addSplitter(intStr.substring(0, this.countBeforeSplitter) + hidden + intStr.substring(intStr.length - this.countBeforeSplitter, intStr.length), splitter);
151 		},
152 
153 		/**
154 		 * For example, ** XXXX
155 		 * @param intStr
156 		 * @param splitter
157 		 * @return {String}
158 		 */
159 		short:	function (intStr, splitter) {
160 		
161 			return '**' + intStr.substring(intStr.length - this.countBeforeSplitter, intStr.length);
162 		}
163 	},

173 	/**
174 	 * Get string like 	'1234 1234 1234 1234'
175 	 * or (by default)	'<span>1234</span><span class="hidden">1234</span><span class="hidden">1234</span><span>1234</span>'
176 	 * format =	null || "full" || "hidden" || "short"
177 	 * splitter =  null || " " || "<span>"
178 	 * @param format
179 	 * @param splitter
180 	 * @return {String}
181 	 */
182 	toHtml:	function (format, splitter) {
183 
184 		/**
185 		 * default format
186 		 */
187 		if (typeof format === 'undefined' || !format) {
188 			format = 'full';
189 		}
190 
191 		/**
192 		 * check format
193 		 */
194 		if (typeof format !== 'string') {
195 			_error('iAccount: Invalid format: "' + format + '"');
196 			return '';
197 		}
198 		
199 		if (typeof this.formatters[format] !== 'function') {
200 			_error('iAccount: Format "' + format + '" is unsupported');
201 			return '';
202 		}
203 		
204 		return this.formatters[format].call(this, this.val, splitter || this.splitter);
205     },

207 	/**
208 	 * check digits count and first digit
209 	 * @return {Boolean}
210 	 */
211 	isCard: function() {
212 
213 		// check digits count
214 		// check first digit
215 		return ([13, 14, 16, 19].indexOf(this.val.length) >= 0 &&
216 		       [3, 4, 5, 6, 7].indexOf(+this.val[0]) >= 0);

248 	/**
249 	 * Check account
250 	 * @return {Boolean}
251 	 */
252 	validate: function() {
253 
254 		if (!this.val){
255 			return false;
256 		}
257 
258 		/**
259 		 * Check type
260 		 */
261 		var isCard;
262 		if( !(isCard = this.isCard()) && !this.isAccount() )
263 			return false;
264 
265 		/**
266 		 * Visa/MC card validation (Luhn Algorithm)
267 		 */
268 		if(isCard){
269 
270 			var sum = 0;
271 			for (var i = 0; i < this.val.length; i++) {
272 
273 				var p = parseInt(this.val[i]);
274 				if (i % 2 == 0) {
275 					p *= 2;
276 					if (p > 9)
277 						p -= 9;
278 				};
279 
280 				sum += p;
281 			};
282 
283 			return sum % 10 == 0
284 		};
285 
286         return true;
287 
288 	},
