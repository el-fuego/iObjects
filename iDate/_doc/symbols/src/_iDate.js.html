 11 iDate = function (val) {
 12 
 13     // copy attributes
 14 	this.attributes = {};
 15 	_this.attributes = this._extendParams(_this.attributes, this.defaults);
 16 
 17 
 18 	/**
 19 	 * param is undefined
 20 	 */
 21 	if (typeof val === 'undefined') {
 22 		this._setDate(new Date());
 23 	} else if (typeof val === 'object' && val instanceof iDate) {
 24 		/**
 25 		 * 	param is iDate
 26 		 */
 27 		this.attributes = this._extendParams({}, val.attributes);
 28 		return this;
 29 	}else if(typeof val === 'number' || val === 'today' || val === 'tomorrow' || val === 'yesterday'
 30         || val === 'now') {
 31 	
 32 		var newDate;
 33         var today = new Date();
 34 
 35         if(val === 'tomorrow'){
 36             newDate = new Date(today.getFullYear(), today.getMonth(),
 37                 today.getDate()+1, 0,0,0,0);
 38         }else if(val === 'yesterday'){
 39             newDate = new Date(today.getFullYear(), today.getMonth(),
 40                 today.getDate()-1, 0,0,0,0);
 41         }else if(val === 'today'){
 42             newDate = new Date(today.getFullYear(), today.getMonth(),
 43                     today.getDate(), 0,0,0,0);
 44         }else if(val === 'now'){
 45             newDate = new Date()
 46         }else if(typeof val === 'number'){
 47             newDate = new Date(+(new Date()) + val*60*60*24*1000);
 48             newDate.setHours(0);
 49             newDate.setMinutes(0);
 50             newDate.setSeconds(0);
 51             newDate.setMilliseconds(0);
 52         }
 53 
 54 		this._setDate(newDate);
 55 	} else if(val instanceof Date){
 56         this._setDate(val);
 57     } else if(typeof val === 'string' ) {
 58 	
 59 		var parsedDate = false;
 60 		var parsedTime = false;
 61 
 62 		/**
 63 		 * parse date
 64 		 */
 65 		for (var i = 0, len = this.parsers.date.length; i < len && !parsedDate && this.parsers.date[i]; i++) {
 66 			parsedDate = this.parsers.date[i].call(this, val);
 67 		}
 68 
 69 		/**
 70 		 * parse time
 71 		 */
 72 		for (var i = 0, len = this.parsers.time.length; i < len && !parsedTime; i++) {
 73 			parsedTime = this.parsers.time[i].call(this, val);
 74 		}
 75 
 76 		/**
 77 		 * check date, f.e. 31.02.1991
 78 		 * @type {Date}
 79 		 */
 80 		var someDate =
 81 			new Date(this.attributes.year.val, (this.attributes.month.val || 1) - 1, this.attributes.date.val || 1);
 82 		
 83 		if (someDate.getDate() != (this.attributes.date.val || 1)
 84 			|| someDate.getMonth()+1 != (this.attributes.month.val || 1)
 85             || someDate.getFullYear() != this.attributes.year.val) {
 86 			
 87 			_error('iDate: Date "'+val+'" is invalid');
 88 
 89             for (var attr in this.attributes) {
 90                 if (typeof this.attributes[attr].val !== 'undefined') {
 91                     this.attributes[attr].val = null;
 92 				}
 93             }
 94         }
 95 		
 96 		if(!parsedDate && !parsedTime) {
 97 			_error('iDate: Check date "'+val+'"');
 98 		}
 99 	}
100 }

300 	/**
301 	 * Получаем номер месяца (от 1 до 12) в зависимости от имени
302 	 * @param monthName
303 	 * @return {null}
304 	 * @private
305 	 */
306     _getMonthIndexByName : function(monthName){
307         var monthMasks = {
308             '1'  : [/янв/gi, /jan/gi],
309             '2'  : [/фев/gi, /feb/gi],
310             '3'  : [/мар/gi, /mar/gi],
311             '4'  : [/апр/gi, /apr/gi],
312             '5'  : [/май/gi, /may/gi],
313             '6'  : [/июн/gi, /jun/gi],
314             '7'  : [/июл/gi, /jul/gi],
315             '8'  : [/авг/gi, /aug/gi],
316             '9'  : [/сен/gi, /sep/gi],
317             '10' : [/окт/gi, /oct/gi],
318             '11' : [/ноя/gi, /nov/gi],
319             '12' : [/дек/gi, /dec/gi]
320         };
321 
322         var monthIndex = null;
323 
324         loop : for(var index in monthMasks){
325             var len = monthMasks[index].length;
326             if(len){
327                 for(var i = 0; i < len; i++){
328                     if(monthMasks[index][i].test(monthName)){
329                         monthIndex = +index;
330                         break loop;
331                     }
332                 }
333             }else{
334                 if(monthMasks[index][len].test(monthName)){
335                     monthIndex = +index;
336                     break loop;
337                 }
338             }
339         };
340 
341         return monthIndex;
342     },

575

581 				return this.get('year') - date.get('year');
582 			case 'months':
583 				return this.get('month') - date.get('month') + (this.get('year') - date.get('year')) * 12;
584 			case 'hours':
585 				return (this.getMilliseconds() - date.getMilliseconds() ) / (60*60*1000);

593 			default:
