 11 /**
 12  * @class iDate
 13  * @param val {iDate|Number|String|Object}
 14  */
 15 iDate = function (val) {
 16 
 17     // copy attributes
 18     this.attributes = {};
 19     this.attributes = this._clone(this.defaults);
 20     var date = new Date();
 21 
 22     /**
 23      * undefined
 24      */
 25     if (val === undefined) {
 26         this._setDate(date);
 27 
 28     } else if (typeof val === 'object' && val instanceof iDate) {
 29         /**
 30          * iDate
 31          */
 32         this.attributes = this._clone(val.attributes);
 33 
 34     } else if (typeof val === 'object' && val !== null) {
 35         /**
 36          * shift parameters
 37          */
 38         var dateType = val.hours || val.minutes || val.seconds || val.ms ? 'now' : 'today';
 39         this.attributes = this._clone((new iDate(dateType)).getShifted(val).attributes);
 40 
 41     } else if (typeof val === 'number') {
 42         /**
 43          * number
 44          */
 45 
 46         // milliseconds
 47         if (val > 60 * 60 * 24 * 1000) {
 48             date = new Date(val);
 49             // days
 50         } else {
 51             date = new Date(+(new Date()) + val * 60 * 60 * 24 * 1000);
 52             date.setHours(0);
 53             date.setMinutes(0);
 54             date.setSeconds(0);
 55             date.setMilliseconds(0);
 56         }
 57         this._setDate(date);
 58 
 59     } else if (val === 'today' || val === 'tomorrow' || val === 'yesterday' || val === 'now') {
 60         /**
 61          * string
 62          */
 63         var today = new Date();
 64 
 65         if (val === 'tomorrow') {
 66             date = new Date(
 67                 today.getFullYear(),
 68                 today.getMonth(),
 69                 today.getDate() + 1,
 70                 0,
 71                 0,
 72                 0,
 73                 0
 74             );
 75         } else if (val === 'yesterday') {
 76             date = new Date(
 77                 today.getFullYear(),
 78                 today.getMonth(),
 79                 today.getDate() - 1,
 80                 0,
 81                 0,
 82                 0,
 83                 0
 84             );
 85         } else if (val === 'today') {
 86             date = new Date(
 87                 today.getFullYear(),
 88                 today.getMonth(),
 89                 today.getDate(),
 90                 0,
 91                 0,
 92                 0,
 93                 0
 94             );
 95         }
 96 
 97         this._setDate(date);
 98     } else if (val instanceof Date) {
 99         this._setDate(val);
100     } else if (typeof val === 'string') {
101         /**
102          * parse date and time
103          */
104 
105         var parsedDate = false;
106         var parsedTime = false;
107         var i, l;
108 
109         for (i = 0, l = this.parsers.date.length; i < l && !parsedDate && this.parsers.date[i]; i++) {
110             parsedDate = this.parsers.date[i].call(this, val);
111         }
112         for (i = 0, l = this.parsers.time.length; i < l && !parsedTime; i++) {
113             parsedTime = this.parsers.time[i].call(this, val);
114         }
115 
116         /**
117          * check date, f.e. 31.02.1991
118          * @type {Date}
119          */
120         var someDate =
121             new Date(this.attributes.year.val, (this.attributes.month.val || 1) - 1, this.attributes.date.val || 1);
122 
123         if (
124             (
125                 someDate.getDate() != (this.attributes.date.val || 1)
126             ) || (
127                 someDate.getMonth() + 1 != (this.attributes.month.val || 1)
128             ) || (
129                 someDate.getFullYear() != this.attributes.year.val
130             )
131         ) {
132 
133             var attr;
134 
135             for (attr in this.attributes) {
136                 if (this.attributes.hasOwnProperty(attr) && this.attributes[attr].val != null) {
137                     this.attributes[attr].val = null;
138                 }
139             }
140         }
141     }
142 };

502     /**
503      * Get full date string
504      * @param format {String|null} null || some kind of 'dd.mm.yyyy HH:MM'
505      * @return {String}
506      */
507     toString: function (format) {
508 
509         if (format == null) {
510             format = this.fullFormat;
511         }
512 
513         var splitted = this._splitFormat(format);
514         var ret = '';
515 
516         while (splitted.length) {
517 
518             var str = splitted.shift();
519             var val = str;
520             var i;
521             var j;
522 
523             /**
524              * elem is a date mask
525              * try to find mask at attributes masks
526              */
527             if (this.formatReg.test(str)) {
528                 for (i in this.attributes) {
529                     if (this.attributes.hasOwnProperty(i) && this.attributes[i].masks) {
530                         for (j in this.attributes[i].masks) {
531                             if (j === str) {
532                                 val = this.attributes[i].masks[j].call(this, this.attributes[i].val);
533                             }
534                         }
535                     }
536                 }
537             }
538 
539             ret += val.replace(/^'([^']+)'|"([^"]+)"$/, "$1$2");
540         }
541 
542         return ret;
543     },

605     /**
606      * Return the difference between dates
607      * @param date {iDate|*} iDate || iDate parameters
608      * @return {Object} {years, months, days, hours, minutes, seconds, ms}
609      */
610     getDifference: function (date) {
611 
612         if (!date) {
613             return null;
614         }
615 
616         if (!(date instanceof iDate)) {
617             date = new window.iDate(date);
618         }
619 
620         return {
621             years:   this.get('year') - date.get('year'),
622             months:  this.get('month') - date.get('month'),
623             days:    this.get('date') - date.get('date'),
624             hours:   this.get('hours') - date.get('hours'),
625             minutes: this.get('minutes') - date.get('minutes'),
626             seconds: this.get('seconds') - date.get('seconds'),
627             ms:      this.get('ms') - date.get('ms')
628         };

631     /**
632      * returns the difference between dates at specified units
633      * @units default is "days"
634      * @param date {iDate|*} iDate || iDate parameters
635      * @param units {String} "years" || "months" || "days" || "hours" || "minutes" || "seconds" || "ms". Default "days"
636      * @return {Float}
637      */
638     getUnitDifference: function (date, units) {
639 
640         if (!date) {
641             return null;
642         }
643 
644         units = units || 'days';
645 
646         if (!(date instanceof iDate)) {
647             date = new iDate(date);
648         }
649 
650         switch (units.toLowerCase()) {
651         case 'years':
652             return this.get('year') - date.get('year') + (
653                 this.get('month') - date.get('month') + (
654                     this.get('date')  - date.get('date') + (
655                         this.get('hours')  - date.get('hours') + (
656                             this.get('minutes')  - date.get('minutes') + (
657                                 this.get('seconds')  - date.get('seconds') + (
658                                     this.get('ms')  - date.get('ms')
659                                 ) / 100
660                             ) / 60
661                         ) / 60
662                     ) / 24
663                 ) / (365 / 12)
664             ) / 12;
665         case 'months':
666             return (this.get('year')  - date.get('year')) * 12 +
667                 this.get('month') - date.get('month') + (
668                     this.get('date')  - date.get('date') + (
669                         this.get('hours')  - date.get('hours') + (
670                             this.get('minutes')  - date.get('minutes') + (
671                                 this.get('seconds')  - date.get('seconds') + (
672                                     this.get('ms')  - date.get('ms')
673                                 ) / 100
674                             ) / 60
675                         ) / 60
676                     ) / 24
677                 ) / (365 / 12);
678         case 'hours':
679             return (this.getMilliseconds() - date.getMilliseconds()) / (60 * 60 * 1000);
680         case 'minutes':
681             return (this.getMilliseconds() - date.getMilliseconds()) / (60 * 1000);
682         case 'seconds':
683             return (this.getMilliseconds() - date.getMilliseconds()) / 1000;
684         case 'ms':
685             return this.getMilliseconds() - date.getMilliseconds();
686         case 'days':
687             return (this.getMilliseconds() - date.getMilliseconds()) / (60 * 60 * 24 * 1000);
688         }
689     },

761             if (this.formatReg.test(str)) {
762                 for (i in this.attributes) {
