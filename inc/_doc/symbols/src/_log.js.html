  5 var _log = function(){
  6 
  7 	if(window.DEBUG && window.console && window.console.info)
  8 		for(var i =0; i<arguments.length; i++)
  9 			window.console.info(arguments[i]);
 10 		
 11 }
 12 
 13 /**
 14  * Функция логирования ошибок.
 15  *  Если выключен режим отладки, функция ничего не выведет.
 16  *  Если послетним аргументом указан флаг === true, то логирование будет выполняться даже при
 17  *  отключенном режиме отладки
 18  */
 19 var _error = function(){
 20 
 21 	if(!window.DEBUG && arguments[arguments.length-1] !== true)
 22 		return;
 23 
 24 	if(window.console && window.console.debug)
 25 		for(var i =0; i<arguments.length; i++)
 26 			window.console.debug(arguments[i])
 27 		
 28 	else if(typeof window.Error === 'function')
 29 		for(var i =0; i<arguments.length; i++)
 30 			new Error(arguments[i])
 31 		
 32 }
 33 
 34 /**
 35  * set DEBUG true on test servers or DEBUG in location
 36  * @type {Boolean}
 37  */
 38 window.DEBUG = ((/^localhost|10\./).test(document.location.host) || (/DEBUG/).test(document.location.href));
