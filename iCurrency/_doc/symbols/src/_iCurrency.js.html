 11 iCurrency = function (curr) {
 12 
 13 	if (typeof curr === 'undefined' || !curr) {
 14 
 15 		/**
 16 		 * search currency
 17 		 * @type {String}
 18 		 */
 19 		this.value = this['default'];
 20 	
 21 		return this;
 22 	}
 23 
 24     if(curr instanceof iCurrency){
 25 
 26         this.value	= curr.value;
 27         return this;
 28     }
 29 
 30 	/**
 31 	 * search currency
 32 	 * @type {String}
 33 	 */
 34 	var c = curr.toUpperCase();
 35 	for (var i in this.currencyList){
 36 		if(this.currencyList[i].test(c)) {
 37 			this.value = i;
 38 			break;
 39 		}
 40 	}
 41 
 42 	if (!this.value) {
 43 		_error('Cant parse currency "'+curr+'"');
 44 	}
 45 };
