1 /**
  2  * @author Knyazevich Denis, Pulyaev Yuriy
  3  * Money Parser|normalizator
  4  * need:    log.js
  5  * use:        a = new iMoney('1.02')
  6  * a.toString()
  7  * @param val
  8  * @param kopeksPointer
  9  * @return {*}
 10  */
 11 iMoney = function (val, kopeksPointer) {
 12 
 13     /**
 14      * set defaults
 15      * @type {null}
 16      */
 17     this['float'] = this['int'] = this['kopeks'] = null;
 18 
 19     /**
 20      * param is undefined
 21      */
 22     if (typeof val === 'undefined' || val === null || val === "") {
 23 
 24         this['float'] = this['int'] = this['kopeks'] = 0;
 25 
 26     } else if (typeof val === 'object' && val instanceof iMoney) {
 27 
 28         this['float'] = val['float'];
 29         this['int'] = val['int'];
 30         this['kopeks'] = val['kopeks'];
 31         return this;
 32 
 33     }
 34 
 35     if (typeof val === 'number') {
 36 
 37         this._setAttributesByFloat(val);
 38 
 39     } else if (typeof val === 'string' && val.replace(/[^0-9,. ]/g, '').length > 0) {
 40 
 41         var arr = null;
 42 
 43         /**
 44          * replace     ect. and whitespaces
 45          * @type {*}
 46          */
 47         val = val.replace(/(&(#[0-9]{3,4}|[a-z]+);)|\s+/gi, '');
 48 
 49 
 50         /**
 51          * use pointer from params if getted
 52          */
 53         if (typeof kopeksPointer === 'string') {
 54 
 55             arr = val.split(kopeksPointer);
 56 
 57             /**
 58              * default pointer
 59              */
 60         } else {
 61 
 62             /**
 63              * 1 234.56
 64              * 1,234.56
 65              * @type {*}
 66              */
 67             arr = val.split('.');
 68 
 69             /**
 70              * 1 234,56
 71              */
 72             if (!arr || arr.length !== 2) {
 73                 arr = val.split(',');
 74             }
 75         }
 76 
 77         /**
 78          * without kopeks
 79          */
 80         if (!arr || arr.length !== 2) {
 81             arr = [val, '0'];
 82         }
 83 
 84         /**
 85          * replace not digits && two "-" :)
 86          * @type {String|XML}
 87          */
 88         arr[0] = arr[0].replace(/[,\. ]/g, '').replace(/\-\-/g, '-');
 89         arr[1] = arr[1].replace(/[,\. ]/g, '');
 90 
 91         /**
 92          * check for errors
 93          */
 94         if (!(/^-?[0-9]+$/).test(arr[0]) || !(/^[0-9]+$/).test(arr[1])) {
 95             return this;
 96         }
 97 
 98         /**
 99          * 0.2 = 0.20
100          */
101         if (arr[1].length === 1) {
102             arr[1] = arr[1] + '0';
103         }
104 
105         var isNegative = arr[0].indexOf('-') >= 0;
106 
107         arr[0] = arr[0].replace(/\-/g, '');
108 
109         var factor = Math.pow(10, arr[1].length || 2);
110 
111         /**
112          * Установка значений.
113          * Делаем дополнительное округление для исправления бага 5.56 -> 5.5600000000000005
114          */
115         this._setAttributesByFloat((isNegative ? -1 : 1) * (
116             Math.round((+arr[0] + (arr[1] / factor)) * factor) / factor
117         ));
118     }
119 };
120 
121 iMoney.prototype = {
122 
123     /**
124      * attributes for info
125      */
126     "float":         null,
127     "int":           null,
128     kopeks:          null,
129     kopeksPointer:   '.',
130     thousandPointer: ' ',
131     htmlTag:         'small',
132 
133 
134     /**
135      * set float, int, kopeks by float value
136      * @param num
137      * @private
138      */
139     _setAttributesByFloat: function (num) {
140 
141         this['float'] = num || 0;
142         this['int'] = parseInt(num) || 0;
143         this['kopeks'] = Math.abs(Math.round(( this['float'] - this['int'] ) * 100)) || 0;
144 
145     },
146 
147     /**
148      * String like '00023'
149      * @param num
150      * @param length
151      * @return {*}
152      * @private
153      */
154     _addZeros: function (num, length) {
155 
156         var ret = num.toString();
157 
158         while (ret.length < length) {
159             ret = '0' + ret;
160         }
161 
162         return ret;
163 
164     },
165 
166     /**
167      * Get full money string
168      * @param kopeksPointer : null || '.'
169      * @param thousandPointer : null || ' '
170      * @param shortFormat : null || true || false
171      * @param toFixed : if true, return 1 522 from 1522.00
172      * @param toABS : false return 1.00 from -1.00
173      * @return {String}
174      */
175     toString: function (kopeksPointer, thousandPointer, shortFormat, toFixed, toABS) {
176         toFixed = (isNaN(toFixed) || toFixed <= 2) ? 2 : toFixed;
177 
178         /**
179          * set default pointers
180          */
181         if (typeof kopeksPointer !== 'string') {
182             kopeksPointer = this.kopeksPointer;
183         }
184 
185         if (typeof thousandPointer !== 'string') {
186             thousandPointer = this.thousandPointer;
187         }
188 
189         if (typeof shortFormat !== 'boolean') {
190             shortFormat = false;
191         }
192 
193         /**
194          * add zeros to kopeks
195          * @type {*|String}
196          */
197         var kopeksStr = (toFixed > 2) ? (this['float'] || 0).toFixed(toFixed).toString().split('.')[1] || '0' : (this['kopeks'] || 0).toString();
198 
199         while (kopeksStr.length < 2) {
200             kopeksStr = '0' + kopeksStr;
201         }
202 
203         /**
204          * add pointer to thousands
205          * @type {String}
206          */
207         var intStr = Math.abs(this['int'] || 0).toString();
208 
209         /**
210          * insert to array every 3 digits
211          * @type {Array}
212          */
213         var intArr = [];
214         while (intStr.length > 0) {
215 
216             if (intStr.length > 3) {
217 
218                 intArr.unshift(intStr.substring(intStr.length - 3, intStr.length));
219 
220                 intStr = intStr.substring(0, intStr.length - 3);
221 
222             } else {
223 
224                 intArr.unshift(intStr.substring(0, intStr.length));
225                 intStr = '';
226             }
227         }
228 
229         return ((this.isNegative() && !toABS) ? '- ' : '') + intArr.join(thousandPointer) + (shortFormat && !this['kopeks'] ? '' : kopeksPointer + kopeksStr);
230     },
231 
232 
233     /**
234      * Get full money string with <htmlTag> in kopeks
235      * @param kopeksPointer : null || '.'
236      * @param thousandPointer : null || ' '
237      * @param htmlTag : null || 'tagName'
238      * @param shortFormat : null || true || false
239      * @param toFixed : true: return 1 522 from 1522.00
240      * @param toABS : false return 1.00 from -1.00
241      * @return {String}
242      */
243     toHtml: function (kopeksPointer, thousandPointer, htmlTag, shortFormat, toFixed, toABS) {
244 
245         /**
246          * set default pointers
247          */
248         if (typeof kopeksPointer !== 'string') {
249             kopeksPointer = this.kopeksPointer;
250         }
251 
252         if (typeof thousandPointer !== 'string') {
253             thousandPointer = this.thousandPointer;
254         }
255 
256         if (typeof htmlTag !== 'string') {
257             htmlTag = this.htmlTag;
258         }
259 
260         if (typeof shortFormat !== 'boolean') {
261             shortFormat = false;
262         }
263 
264         /**
265          * split to [amount, kopeks]
266          * @type {Array}
267          */
268         var arr = this.toString(kopeksPointer, thousandPointer, shortFormat, toFixed, toABS).replace(/ /g, ' ').split(kopeksPointer);
269 
270         /**
271          * add tag to kopeks
272          * @type {String}
273          */
274         arr[1] = '<' + htmlTag + '>' + arr[1] + '</' + htmlTag + '>';
275 
276         return arr.join(kopeksPointer);
277     },
278 
279     /**
280      * Get Float, Integer, Number
281      * @return {null|int}
282      */
283     toFloat:   function () {
284         return (this['float'] || 0);
285     },
286     toNumber:  function () {
287         return (this['float'] || 0);
288     },
289     toInteger: function () {
290         return (this['int'] || 0);
291     },
292     toInt:     function () {
293         return (this['int'] || 0);
294     },
295 
296     isNull: function () {
297         return  (this['float'] == 0 || this['float'] == null);
298     },
299 
300     isNegative: function () {
301         return  this.toFloat() < 0;
302     },
303 
304     getAbs: function () {
305         return new iMoney(Math.abs(this.toFloat()));
306     },
307 
308     getInverted: function () {
309         return new iMoney(-1 * this.toFloat());
310     },
311 
312     /**
313      * Check data
314      * @return {Boolean}
315      */
316     validate: function () {
317         return this['float'] !== null;
318     },
319 
320     /**
321      *
322      * @param digits
323      */
324     toFixed: function (digits) {
325         digits = +(digits || 0);
326 
327         var factor = Math.pow(10, digits);
328 
329         return Math.round(this['float'] * factor) / factor;
330     }
