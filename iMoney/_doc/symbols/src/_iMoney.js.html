1 /**
  2  * @author Knyazevich Denis, Pulyaev Yuriy
  3  * Money Parser|normalizator
  4  * need:    log.js
  5  * use:        a = new iMoney('1.02')
  6  * a.toString()
  7  * @param val
  8  * @param kopeksPointer
  9  * @return {*}
 10  */
 11 iMoney = function (val, kopeksPointer) {
 12 
 13 	/**
 14 	 * set defaults
 15 	 * @type {null}
 16 	 */
 17 	this['float'] = this['int'] = this['kopeks'] = null;
 18 
 19 	/**
 20 	 * param is undefined
 21 	 */
 22 	if (typeof val === 'undefined' || val === null || val === "") {
 23 
 24 		this['float'] = this['int'] = this['kopeks'] = 0;
 25 
 26 	} else if (typeof val === 'object' && val instanceof iMoney) {
 27 
 28 		this['float'] = val['float'];
 29 		this['int'] = val['int'];
 30 		this['kopeks'] = val['kopeks'];
 31 		return this;
 32 
 33 	} else if (typeof val === 'number') {
 34 
 35 		this._setAttributesByFloat(val);
 36 
 37 	} else if (typeof val === 'string' && val.replace(/[^0-9,. ]/g, '').length > 0) {
 38 
 39 		var arr = null;
 40 
 41 		/**
 42 		 * replace     ect. and whitespaces
 43 		 * @type {*}
 44 		 */
 45 		var val = val.replace(/(&(#[0-9]{3,4}|[a-z]+);)|\s+/gi, '');
 46 
 47 
 48 		/**
 49 		 * check for errors
 50 		 */
 51 		if (!(/^-?[0-9., ]+$/).test(val)) {
 52 
 53 			_error('Check amount "' + val + '"');
 54 		}
 55 
 56 		/**
 57 		 * use pointer from params if getted
 58 		 */
 59 		if (typeof kopeksPointer === 'string') {
 60 
 61 			arr = val.split(kopeksPointer);
 62 
 63 			/**
 64 			 * default pointer
 65 			 */
 66 		} else {
 67 
 68 			/**
 69 			 * 1 234.56
 70 			 * 1,234.56
 71 			 * @type {*}
 72 			 */
 73 			arr = val.split('.');
 74 
 75 			/**
 76 			 * 1 234,56
 77 			 */
 78 			if (!arr || arr.length !== 2)
 79 				arr = val.split(',');
 80 		}
 81 
 82 		/**
 83 		 * without kopeks
 84 		 */
 85 		if (!arr || arr.length !== 2)
 86 			arr = [val, '0'];
 87 
 88 		/**
 89 		 * replace not digits && two "-" :)
 90 		 * @type {String|XML}
 91 		 */
 92 		arr[0] = arr[0].replace(/[,\. ]/g, '').replace(/\-\-/g, '-');
 93 		arr[1] = arr[1].replace(/[,\. ]/g, '');
 94 
 95 		/**
 96 		 * check for errors
 97 		 */
 98 		if (!(/^-?[0-9]+$/).test(arr[0]) || !(/^[0-9]+$/).test(arr[1])) {
 99 			return this;
100 		}
101 
102 		/**
103 		 * 0.2 = 0.20
104 		 */
105 		if (arr[1].length === 1)
106 			arr[1] = arr[1] + '0';
107 
108 		var isNegative = arr[0].indexOf('-') >= 0;
109 
110 		arr[0] = arr[0].replace(/\-/g, '');
111 
112 		// set val
113 		this._setAttributesByFloat(
114 			(isNegative ? -1 : 1) * ( +(arr[0]) + +(arr[1]) / Math.pow(10, arr[1].length || 2) ));
115 	}
116 }
117 
118 iMoney.prototype = {
119 
120 	/**
121 	 * attributes for info
122 	 */
123 	float:                 null,
124 	int:                   null,
125 	kopeks:                null,
126 	kopeksPointer:         '.',
127 	thousandPointer:       ' ',
128 	htmlTag:               'small',
129 
130 
131 	/**
132 	 * set float, int, kopeks by float value
133 	 * @param num
134 	 * @private
135 	 */
136 	_setAttributesByFloat: function (num) {
137 
138 		this['float'] = num || 0;
139 		this['int'] = parseInt(num) || 0;
140 		this['kopeks'] = Math.abs(Math.round(( this['float'] - this['int'] ) * 100)) || 0;
141 
142 	},
143 
144 	/**
145 	 * String like '00023'
146 	 * @param num
147 	 * @param length
148 	 * @return {*}
149 	 * @private
150 	 */
151 	_addZeros: function (num, length) {
152 
153 		var ret = num.toString();
154 
155 		while (ret.length < length)
156 			ret = '0' + ret;
157 
158 		return ret;
159 
160 	},
161 
162 	/**
163 	 * Get full money string
164 	 * @param kopeksPointer : null || '.'
165 	 * @param thousandPointer : null || ' '
166 	 * @param shortFormat : null || true || false
167 	 * @param toFixed : if true, return 1 522 from 1522.00
168 	 * @return {String}
169 	 */
170 	toString: function (kopeksPointer, thousandPointer, shortFormat, toFixed) {
171 		toFixed = (isNaN(toFixed) || toFixed <= 2) ? 2 : toFixed;
172 
173 		/**
174 		 * set default pointers
175 		 */
176 		if (typeof kopeksPointer !== 'string')
177 			kopeksPointer = this.kopeksPointer;
178 
179 		if (typeof thousandPointer !== 'string')
180 			thousandPointer = this.thousandPointer;
181 
182 		if (typeof shortFormat !== 'boolean')
183 			shortFormat = false;
184 
185 		/**
186 		 * add zeros to kopeks
187 		 * @type {*|String}
188 		 */
189 		var kopeksStr = (toFixed > 2) ?
190 			(this['float'] || 0).toFixed(toFixed).toString().split('.')[1] || '0' :
191 			(this['kopeks'] || 0).toString();
192 
193 		while (kopeksStr.length < 2)
194 			kopeksStr = '0' + kopeksStr;
195 
196 		/**
197 		 * add pointer to thousands
198 		 * @type {String}
199 		 */
200 		var intStr = Math.abs(this['int'] || 0).toString();
201 
202 		/**
203 		 * insert to array every 3 digits
204 		 * @type {Array}
205 		 */
206 		var intArr = [];
207 		while (intStr.length > 0) {
208 
209 			if (intStr.length > 3) {
210 
211 				intArr.unshift(intStr.substring(intStr.length - 3, intStr.length));
212 
213 				intStr = intStr.substring(0, intStr.length - 3);
214 
215 			} else {
216 
217 				intArr.unshift(intStr.substring(0, intStr.length));
218 				intStr = '';
219 			}
220 		}
221 
222 		return (this.isNegative() ? '- ' : '') +
223 			intArr.join(thousandPointer) + (shortFormat && !this['kopeks'] ? '' : kopeksPointer + kopeksStr);
224 	},
225 
226 
227 	/**
228 	 * Get full money string with <htmlTag> in kopeks
229 	 * @param kopeksPointer : null || '.'
230 	 * @param thousandPointer : null || ' '
231 	 * @param htmlTag : null || 'tagName'
232 	 * @param shortFormat : null || true || false
233 	 * @param toFixed : true: return 1 522 from 1522.00
234 	 * @return {String}
235 	 */
236 	toHtml: function (kopeksPointer, thousandPointer, htmlTag, shortFormat, toFixed) {
237 
238 		/**
239 		 * set default pointers
240 		 */
241 		if (typeof kopeksPointer !== 'string')
242 			kopeksPointer = this.kopeksPointer;
243 
244 		if (typeof thousandPointer !== 'string')
245 			thousandPointer = this.thousandPointer;
246 
247 		if (typeof htmlTag !== 'string')
248 			htmlTag = this.htmlTag;
249 
250 		if (typeof shortFormat !== 'boolean')
251 			shortFormat = false;
252 
253 		/**
254 		 * split to [amount, kopeks]
255 		 * @type {Array}
256 		 */
257 		var arr = this.toString(kopeksPointer, thousandPointer, shortFormat, toFixed).replace(/ /g, ' ').split(kopeksPointer);
258 
259 		/**
260 		 * add tag to kopeks
261 		 * @type {String}
262 		 */
263 		arr[1] = '<' + htmlTag + '>' + arr[1] + '</' + htmlTag + '>';
264 
265 		return arr.join(kopeksPointer);
266 	},
267 
268 	/**
269 	 * Get Float, Integer, Number
270 	 * @return {null|int}
271 	 */
272 	toFloat: function () {
273 		return (this.float || 0);
274 	},
275 	toNumber: function () {
276 		return (this.float || 0);
277 	},
278 	toInteger: function () {
279 		return (this['int'] || 0);
280 	},
281 	toInt: function () {
282 		return (this['int'] || 0);
283 	},
284 
285 	isNull: function () {
286 		return  (this['float'] == 0 || this['float'] == null);
287 	},
288 
289 	isNegative: function () {
290 		return  this.toFloat() < 0;
291 	},
292 
293 	getAbs:   function () {
294 		return new iMoney(Math.abs(this.toFloat()));
295 	},
296 
297 	/**
298 	 * Check data
299 	 * @return {Boolean}
300 	 */
301 	validate: function () {
302 		return this['float'] !== null;
303 	}
304 };
